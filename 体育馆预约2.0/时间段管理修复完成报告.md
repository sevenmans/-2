async clearTimeSlotCache(venueId, date) {
  // 1. 清理Store中的时间段数据
  this.timeSlots = []
  this.timeSlotsLoading = false
  
  // 2. 清理时间段管理器缓存
  if (this.timeSlotManager) {
    this.timeSlotManager.clearCache()
  }
  
  // 3. 清理本地存储缓存
  const cacheKeys = uni.getStorageInfoSync().keys || []
  cacheKeys.forEach(key => {
    if (key.includes('timeslot_') || key.includes('venue_') || 
        key.includes('booking_') || key.includes('cache_')) {
      uni.removeStorageSync(key)
    }
  })
  
  // 4. 清理内存中的临时数据
  this.tempTimeSlotData = null
  this.lastRefreshTime = null
}
