## 2. 时间段生成机制

### 2.1 核心生成方法

**位置**: `TimeSlotService.generateTimeSlotsForVenue()`

**生成规则**:
1. **时间间隔**: 按30分钟为单位生成时间段
2. **时间范围**: 根据场馆的营业时间（openTime - closeTime）
3. **默认时间**: 如果场馆未设置营业时间，默认为 09:00-22:00
4. **价格计算**: 每个30分钟时间段价格 = 场馆小时价格 / 2
5. **重复检查**: 生成前检查是否已存在时间段，避免重复生成

```java
@Transactional
public List<TimeSlot> generateTimeSlotsForVenue(Long venueId, LocalDate date) {
    // 1. 验证场馆存在性
    Optional<Venue> venueOpt = venueRepository.findById(venueId);
    if (!venueOpt.isPresent()) {
        throw new RuntimeException("场馆不存在，ID: " + venueId);
    }
    
    // 2. 获取场馆信息
    Venue venue = venueOpt.get();
    LocalTime openTime = venue.getOpenTime() != null ? venue.getOpenTime() : LocalTime.of(9, 0);
    LocalTime closeTime = venue.getCloseTime() != null ? venue.getCloseTime() : LocalTime.of(22, 0);
    Double hourlyPrice = venue.getPrice();
    
    // 3. 检查是否已生成
    List<TimeSlot> existingSlots = timeSlotRepository.findByVenueIdAndDate(venueId, date);
    if (!existingSlots.isEmpty()) {
        return existingSlots; // 已存在，直接返回
    }
    
    // 4. 生成时间段
    List<TimeSlot> slots = new ArrayList<>();
    LocalTime currentTime = openTime;
    
    while (currentTime.isBefore(closeTime)) {
        LocalTime endTime = currentTime.plusMinutes(30);
        if (endTime.isAfter(closeTime)) {
            break; // 超出营业时间，停止生成
        }
        
        // 计算30分钟价格
        Double slotPrice = hourlyPrice / 2;
        
        TimeSlot slot = new TimeSlot(venueId, date, currentTime, endTime, slotPrice);
        slots.add(slot);
        
        currentTime = endTime; // 移动到下一个时间段
    }
    
    // 5. 批量保存
    return timeSlotRepository.saveAll(slots);
}
```

// 位置: store/modules/venue.js
processTimeSlotStatus(timeSlots, date) {
    const now = new Date()
    const currentDate = now.toISOString().split('T')[0]
    
    return timeSlots.map(slot => {
        // 1. 状态字段标准化
        if (slot.available !== undefined) {
            slot.status = slot.available ? 'AVAILABLE' : 'OCCUPIED'
        }
        
        // 2. 过期检查（仅当天）
        if (date === currentDate && slot.status === 'AVAILABLE') {
            const slotStartTime = slot.startTime.split(':').map(Number)
            const slotStartMinutes = slotStartTime[0] * 60 + slotStartTime[1]
            const currentTime = now.getHours() * 60 + now.getMinutes()
            
            if (slotStartMinutes <= currentTime) {
                slot.status = 'EXPIRED'
                slot.originalStatus = 'AVAILABLE'
                slot.disabled = true
            }
        }
        
        // 3. 状态映射和UI属性设置
        switch (slot.status) {
            case 'AVAILABLE':
                slot.disabled = false
                slot.buttonText = '预约'
                slot.buttonClass = 'available'
                break
            case 'OCCUPIED':
            case 'BOOKED':
            case 'RESERVED':
                slot.disabled = true
                slot.buttonText = '已占用'
                slot.buttonClass = 'occupied'
                break
            case 'MAINTENANCE':
                slot.disabled = true
                slot.buttonText = '维护中'
                slot.buttonClass = 'maintenance'
                break
            case 'SHARING':
                slot.disabled = false
                slot.buttonText = '拼场'
                slot.buttonClass = 'sharing'
                break
            case 'EXPIRED':
                slot.disabled = true
                slot.buttonText = '已过期'
                slot.buttonClass = 'expired'
                break
            default:
                slot.disabled = true
                slot.buttonText = '不可用'
                slot.buttonClass = 'unavailable'
        }
        
        return slot
    })
}

// 位置: store/modules/venue.js
class SmartRefreshManager {
    constructor() {
        this.refreshQueue = new Map() // 刷新队列
        this.lastRefreshTimes = new Map() // 上次刷新时间记录
        this.refreshIntervals = {
            high: 30000,    // 高频刷新：30秒（当前查看的日期）
            medium: 60000,  // 中频刷新：1分钟（相邻日期）
            low: 300000     // 低频刷新：5分钟（其他日期）
        }
    }
    
    // 根据日期重要性确定刷新频率
    getRefreshInterval(date) {
        const today = new Date().toISOString().split('T')[0]
        const tomorrow = new Date(Date.now() + 86400000).toISOString().split('T')[0]
        const yesterday = new Date(Date.now() - 86400000).toISOString().split('T')[0]
        
        if (date === today) {
            return this.refreshIntervals.high // 当天高频刷新
        } else if (date === tomorrow || date === yesterday) {
            return this.refreshIntervals.medium // 相邻日期中频刷新
        } else {
            return this.refreshIntervals.low // 其他日期低频刷新
        }
    }
    
    // 智能刷新调度
    scheduleRefresh(venueId, date, priority = 'medium') {
        const key = `${venueId}_${date}`
        const interval = this.getRefreshInterval(date)
        const lastRefresh = this.lastRefreshTimes.get(key) || 0
        const now = Date.now()
        
        // 检查是否需要刷新
        if (now - lastRefresh >= interval) {
            this.executeRefresh(venueId, date)
            this.lastRefreshTimes.set(key, now)
        } else {
            // 加入刷新队列，延后执行
            const remainingTime = interval - (now - lastRefresh)
            setTimeout(() => {
                this.executeRefresh(venueId, date)
                this.lastRefreshTimes.set(key, Date.now())
            }, remainingTime)
        }
    }
    
    async executeRefresh(venueId, date) {
        try {
            await store.dispatch('venue/getVenueTimeSlots', { venueId, date })
            console.log(`智能刷新完成: 场馆${venueId}, 日期${date}`)
        } catch (error) {
            console.error(`智能刷新失败: 场馆${venueId}, 日期${date}`, error)
        }
    }
}

// 位置: utils/websocket.js
class TimeSlotWebSocket {
    constructor() {
        this.ws = null
        this.reconnectAttempts = 0
        this.maxReconnectAttempts = 5
        this.reconnectInterval = 3000
        this.heartbeatInterval = 30000
        this.heartbeatTimer = null
    }
    
    // 连接WebSocket
    connect() {
        try {
            const wsUrl = `${config.WS_BASE_URL}/timeslot-updates`
            this.ws = new WebSocket(wsUrl)
            
            this.ws.onopen = this.onOpen.bind(this)
            this.ws.onmessage = this.onMessage.bind(this)
            this.ws.onclose = this.onClose.bind(this)
            this.ws.onerror = this.onError.bind(this)
            
        } catch (error) {
            console.error('WebSocket连接失败:', error)
            this.scheduleReconnect()
        }
    }
    
    onOpen() {
        console.log('WebSocket连接已建立')
        this.reconnectAttempts = 0
        this.startHeartbeat()
        
        // 订阅当前用户关注的场馆
        this.subscribeToVenues()
    }
    
    onMessage(event) {
        try {
            const data = JSON.parse(event.data)
            
            switch (data.type) {
                case 'TIMESLOT_STATUS_CHANGE':
                    this.handleTimeSlotStatusChange(data.payload)
                    break
                case 'TIMESLOT_GENERATED':
                    this.handleTimeSlotGenerated(data.payload)
                    break
                case 'HEARTBEAT':
                    // 心跳响应
                    break
                default:
                    console.log('未知消息类型:', data.type)
            }
        } catch (error) {
            console.error('WebSocket消息解析失败:', error)
        }
    }
    
    // 处理时间段状态变更
    handleTimeSlotStatusChange(payload) {
        const { venueId, date, timeSlotId, oldStatus, newStatus } = payload
        
        // 更新Vuex状态
        store.dispatch('venue/updateTimeSlotStatus', {
            venueId,
            date,
            timeSlotId,
            newStatus
        })
        
        // 显示通知
        if (this.isCurrentlyViewing(venueId, date)) {
            uni.showToast({
                title: `时间段状态已更新: ${oldStatus} → ${newStatus}`,
                icon: 'none',
                duration: 2000
            })
        }
    }
    
    // 处理新生成的时间段
    handleTimeSlotGenerated(payload) {
        const { venueId, date, timeSlots } = payload
        
        // 刷新对应的时间段数据
        store.dispatch('venue/getVenueTimeSlots', { venueId, date })
        
        console.log(`场馆${venueId}在${date}新生成了${timeSlots.length}个时间段`)
    }
    
    // 订阅场馆更新
    subscribeToVenues() {
        const currentUser = store.getters['user/currentUser']
        if (currentUser && this.ws.readyState === WebSocket.OPEN) {
            this.ws.send(JSON.stringify({
                type: 'SUBSCRIBE',
                userId: currentUser.id,
                venues: store.getters['venue/favoriteVenues'] // 用户收藏的场馆
            }))
        }
    }
    
    // 检查是否正在查看该场馆和日期
    isCurrentlyViewing(venueId, date) {
        const currentRoute = getCurrentPages()
        const currentPage = currentRoute[currentRoute.length - 1]
        
        return currentPage.route.includes('venue/detail') &&
               currentPage.options.venueId == venueId &&
               store.getters['venue/selectedDate'] === date
    }
    
    // 心跳机制
    startHeartbeat() {
        this.heartbeatTimer = setInterval(() => {
            if (this.ws.readyState === WebSocket.OPEN) {
                this.ws.send(JSON.stringify({ type: 'HEARTBEAT' }))
            }
        }, this.heartbeatInterval)
    }
    
    onClose() {
        console.log('WebSocket连接已关闭')
        this.clearHeartbeat()
        this.scheduleReconnect()
    }
    
    onError(error) {
        console.error('WebSocket错误:', error)
    }
    
    // 重连机制
    scheduleReconnect() {
        if (this.reconnectAttempts < this.maxReconnectAttempts) {
            this.reconnectAttempts++
            console.log(`尝试重连WebSocket (${this.reconnectAttempts}/${this.maxReconnectAttempts})`)
            
            setTimeout(() => {
                this.connect()
            }, this.reconnectInterval * this.reconnectAttempts)
        } else {
            console.error('WebSocket重连失败，已达到最大重试次数')
        }
    }
    
    clearHeartbeat() {
        if (this.heartbeatTimer) {
            clearInterval(this.heartbeatTimer)
            this.heartbeatTimer = null
        }
    }
    
    disconnect() {
        this.clearHeartbeat()

// 位置: utils/cache.js
class TimeSlotCacheManager {
    constructor() {
        this.cache = new Map()
        this.cacheExpiry = new Map()
        this.defaultTTL = 300000 // 5分钟默认过期时间
    }
    
    // 设置缓存
    set(key, data, ttl = this.defaultTTL) {
        this.cache.set(key, data)
        this.cacheExpiry.set(key, Date.now() + ttl)
        
        // 设置过期清理
        setTimeout(() => {
            this.delete(key)
        }, ttl)
    }
    
    // 获取缓存
    get(key) {
        const expiry = this.cacheExpiry.get(key)
        if (expiry && Date.now() > expiry) {
            this.delete(key)
            return null
        }
        return this.cache.get(key)
    }
    
    // 删除缓存
    delete(key) {
        this.cache.delete(key)
        this.cacheExpiry.delete(key)
    }
    
    // 清空所有缓存
    clear() {
        this.cache.clear()
        this.cacheExpiry.clear()
    }
    
    // 智能缓存刷新
    async smartRefresh(venueId, date, forceRefresh = false) {
        const cacheKey = `timeslots_${venueId}_${date}`
        const cachedData = this.get(cacheKey)
        
        // 如果有缓存且不强制刷新，返回缓存数据
        if (cachedData && !forceRefresh) {
            return cachedData
        }
        
        try {
            // 从API获取最新数据
            const freshData = await venueApi.getVenueTimeSlots(venueId, date)
            
            // 根据日期设置不同的缓存时间
            const today = new Date().toISOString().split('T')[0]
            let ttl = this.defaultTTL
            
            if (date === today) {
                ttl = 60000 // 当天数据1分钟过期
            } else if (date < today) {
                ttl = 3600000 // 历史数据1小时过期
            } else {
                ttl = 300000 // 未来数据5分钟过期
            }
            
            this.set(cacheKey, freshData, ttl)
            return freshData
            
        } catch (error) {
            // 如果API失败但有缓存，返回过期缓存
            if (cachedData) {
                console.warn('API请求失败，返回过期缓存数据')
                return cachedData
            }
            throw error
        }
    }
}

@Component
public class TimeSlotRefreshLock {
    
    @Autowired
    private RedisTemplate<String, String> redisTemplate;
    
    public boolean acquireRefreshLock(String venueId, String date) {
        String lockKey = "timeslot:refresh:" + venueId + ":" + date;
        String lockValue = UUID.randomUUID().toString();
        
        Boolean success = redisTemplate.opsForValue()
            .setIfAbsent(lockKey, lockValue, Duration.ofMinutes(5));
        
        return Boolean.TRUE.equals(success);
    }
    
    public void releaseRefreshLock(String venueId, String date) {
        String lockKey = "timeslot:refresh:" + venueId + ":" + date;
        redisTemplate.delete(lockKey);
    }
}
