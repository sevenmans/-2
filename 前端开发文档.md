## 📖 项目概述

### 项目简介
体育馆预约系统是一个基于微信小程序的体育场馆预约平台，用户可以通过小程序查看场馆信息、预约时间段、参与拼场等功能。

### 核心功能
- 🏟️ **场馆管理**: 查看场馆列表、详情、时间段
- 👤 **用户系统**: 注册、登录、个人信息管理
- 📅 **预约系统**: 创建预约、查看预约记录、取消预约
- 🤝 **拼场功能**: 创建拼场、加入拼场、拼场管理
- 💰 **订单管理**: 订单创建、支付、状态跟踪
- 📢 **公告系统**: 查看系统公告和通知

---

## 🛠️ 技术栈与环境

### 前端技术栈
- **框架**: uni-app (Vue 3)
- **开发工具**: HBuilderX
- **调试工具**: 微信小程序开发助手
- **UI框架**: uni-ui
- **状态管理**: Vuex 4
- **网络请求**: uni.request
- **图标**: uni-icons

### 开发环境要求
- HBuilderX 3.8+
- 微信开发者工具
- Node.js 16+
- 微信小程序账号

### 后端接口
- **基础URL**: `http://localhost:8080/api`
- **认证方式**: JWT Token
- **数据格式**: JSON

## 🔌 API接口详细说明

### 基础配置

```javascript
// utils/request.js
const BASE_URL = 'http://localhost:8080/api'
const TIMEOUT = 10000

// 请求拦截器
const request = (options) => {
  return new Promise((resolve, reject) => {
    // 添加认证头
    const token = uni.getStorageSync('token')
    if (token) {
      options.header = {
        ...options.header,
        'Authorization': `Bearer ${token}`
      }
    }
    
    uni.request({
      url: BASE_URL + options.url,
      method: options.method || 'GET',
      data: options.data,
      header: {
        'Content-Type': 'application/json',
        ...options.header
      },
      timeout: TIMEOUT,
      success: (res) => {
        if (res.statusCode === 200) {
          resolve(res.data)
        } else {
          reject(res)
        }
      },
      fail: reject
    })
  })
}
```

```vue
      venueTypes: [
        { label: '全部', value: '' },

        { label: '羽毛球', value: '羽毛球' },

      if (this.loading) return

        if (reset) {
          this.page = 1
          await this.resetVenueList()
        }
        
        await this.getVenueList({
          page: this.page,
          type: this.selectedType,
          keyword: this.searchKeyword
        })
```

## 🌐 网络请求封装

### 1. 请求工具 (utils/request.js)

```javascript
import { getToken, removeToken } from './auth'
import { showToast, showLoading, hideLoading } from './ui'

// 基础配置
const BASE_URL = 'http://localhost:8080/api'
const TIMEOUT = 10000

// 请求拦截器
class Request {
  constructor() {
    this.baseURL = BASE_URL
    this.timeout = TIMEOUT
  }
  
  // 通用请求方法
  request(options) {
    return new Promise((resolve, reject) => {
      // 添加认证头
      const token = getToken()
      if (token) {
        options.header = {
          ...options.header,
          'Authorization': `Bearer ${token}`
        }
      }
      
      // 显示加载状态
      if (options.loading !== false) {
        showLoading(options.loadingText || '加载中...')
      }
      
      uni.request({
        url: this.baseURL + options.url,
        method: options.method || 'GET',
        data: options.data,
        header: {
          'Content-Type': 'application/json',
          ...options.header
        },
        timeout: this.timeout,
        success: (res) => {
          hideLoading()
          
          if (res.statusCode === 200) {
            resolve(res.data)
          } else if (res.statusCode === 401) {
            // token过期，清除登录状态
            this.handleTokenExpired()
            reject(new Error('登录已过期，请重新登录'))
          } else {
            const error = new Error(res.data?.message || '请求失败')
            error.statusCode = res.statusCode
            reject(error)
          }
        },
        fail: (err) => {
          hideLoading()
          
          let message = '网络连接失败'
          if (err.errMsg) {
            if (err.errMsg.includes('timeout')) {
              message = '请求超时'
            } else if (err.errMsg.includes('fail')) {
              message = '网络连接失败'
            }
          }
          
          const error = new Error(message)
          error.errMsg = err.errMsg
          reject(error)
        }
      })
    })
  }
  
  // 处理token过期
  handleTokenExpired() {
    removeToken()
    
    // 跳转到登录页
    uni.reLaunch({
      url: '/pages/user/login'
    })
    
    showToast('登录已过期，请重新登录')
  }
  
  // GET请求
  get(url, params = {}, options = {}) {
    return this.request({
      url,
      method: 'GET',
      data: params,
      ...options
    })
  }
  
  // POST请求
  post(url, data = {}, options = {}) {
    return this.request({
      url,
      method: 'POST',
      data,
      ...options
    })
  }
  
  // PUT请求
  put(url, data = {}, options = {}) {
    return this.request({
      url,
      method: 'PUT',
      data,
      ...options
    })
  }
  
  // DELETE请求
  delete(url, data = {}, options = {}) {
    return this.request({
      url,
      method: 'DELETE',
      data,
      ...options
    })
  }
  
  // 文件上传
  upload(url, filePath, options = {}) {
    return new Promise((resolve, reject) => {
      const token = getToken()
      
      showLoading('上传中...')
      
      uni.uploadFile({
        url: this.baseURL + url,
        filePath,
        name: options.name || 'file',
        header: {
          'Authorization': token ? `Bearer ${token}` : '',
          ...options.header
        },
        formData: options.formData,
        success: (res) => {
          hideLoading()
          
          if (res.statusCode === 200) {
            try {
              const data = JSON.parse(res.data)
              resolve(data)
            } catch (e) {
              resolve(res.data)
            }
          } else {
            reject(new Error('上传失败'))
          }
        },
        fail: (err) => {
          hideLoading()
          reject(new Error('上传失败'))
        }
      })
    })
  }
}

// 创建实例
const request = new Request()

export default request
```

```javascript
    }
  }
  
  return true
}
```

```javascript
  if (!userInfo || !userInfo.roles) {
    return false
  }
  return userInfo.roles.includes(role)
}

export function hasPermission(permission) {
  const userInfo = getUserInfo()
  if (!userInfo || !userInfo.permissions) {
    return false
  }
  return userInfo.permissions.includes(permission)
}

// 检查是否为管理员
export function isAdmin() {
  return hasRole('ROLE_ADMIN')
}

// 检查是否为普通用户
export function isUser() {

}
```

```javascript
// API调用错误处理
async function fetchData() {
  try {
    const response = await api.getData()
    return response
  } catch (error) {
    console.error('获取数据失败:', error)
    
    // 根据错误类型处理
    if (error.statusCode === 401) {
      // 未授权，跳转登录
      uni.navigateTo({ url: '/pages/user/login' })
    } else if (error.statusCode === 403) {
      // 权限不足
      uni.showToast({ title: '权限不足', icon: 'error' })
    } else {
      // 其他错误
      uni.showToast({ title: error.message || '操作失败', icon: 'error' })
    }
    
    throw error
  }
}

// 表单验证错误处理
function validateForm(formData) {
  const errors = []
  
  if (!formData.username) {
    errors.push('用户名不能为空')
  }
  
  if (!formData.password) {
    errors.push('密码不能为空')
  }
  
  if (errors.length > 0) {
    uni.showToast({ title: errors[0], icon: 'none' })
    return false
  }
  
  return true
```

```javascript
// utils/helpers.js

// 日期格式化
export function formatDate(date, format = 'YYYY-MM-DD') {
  const d = new Date(date)
  const year = d.getFullYear()
  const month = String(d.getMonth() + 1).padStart(2, '0')
  const day = String(d.getDate()).padStart(2, '0')
  
  return format
    .replace('YYYY', year)
    .replace('MM', month)
    .replace('DD', day)
}

// 价格格式化
export function formatPrice(price) {
  return `¥${Number(price).toFixed(2)}`
}

// 手机号脱敏
export function maskPhone(phone) {
  return phone.replace(/(\d{3})\d{4}(\d{4})/, '$1****$2')
}

// 防抖函数
export function debounce(func, wait) {
  let timeout
  return function executedFunction(...args) {
    const later = () => {
      clearTimeout(timeout)
      func(...args)
    }
    clearTimeout(timeout)
    timeout = setTimeout(later, wait)
  }
}

// 节流函数
export function throttle(func, limit) {
  let inThrottle
  return function() {
    const args = arguments
    const context = this
    if (!inThrottle) {
      func.apply(context, args)
      inThrottle = true
      setTimeout(() => inThrottle = false, limit)
    }
  }
}
```
